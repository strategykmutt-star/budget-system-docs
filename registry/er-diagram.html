อ้างอิงหน้าจอจริงจากเอกสารอบรม 11 มิ.ย. 2569 (โปรแกรม Bis65xxxx / Lao66b101) — ร่างสำหรับพัฒนาระบบ

แผนภาพความสัมพันธ์

หลักการออกแบบ

#หลักการเหตุผลจากเอกสาร
1ผูกปีงบประมาณทุกตาราง (fiscal_year)ทุกจอกรองด้วย "ปี"; ปุ่ม "ค้นหาทะเบียนตามปี" = ยกยอดข้ามปี
2Effective dating start_date/end_date (NULL = เปิดอยู่)จอแสดงวันที่เริ่ม–สิ้นสุด, 99/99/99 = ไม่มีวันสิ้นสุด
3Soft-close ก่อน hard-delete (row_status)มีปุ่ม ปิด/เปิด-ปิด แยกจาก ลบ; ลบ "ตามสิทธิ์"
4ห้ามลบเมื่อมี child (FK RESTRICT)รักษา integrity ของสายโซ่ทะเบียน
5แยกสิทธิ์ตามระบบ (system_type)e-Budgeting แก้เอง / BBL อ่านอย่างเดียว
6รายการเป็นต้นไม้ (parent_item_id + item_level)จอ "ทะเบียนรายการ (ระดับชั้น)"

คอลัมน์ร่วม (ชั้น 3–8 ทุกตาราง)

fiscal_year   smallint     NOT NULL REFERENCES ref_fiscal_year,

created_by    varchar(50)  NOT NULL,
created_at    timestamptz  NOT NULL DEFAULT now(),
updated_by    varchar(50)  NULL,
updated_at    timestamptz  NULL

DDL ร่าง (PostgreSQL-style)

-- ===== ตารางอ้างอิง =====
CREATE TABLE ref_fiscal_year (
  fiscal_year   smallint PRIMARY KEY,            -- 70
  period_status varchar(10) NOT NULL );          -- OPEN | CLOSED

CREATE TABLE ref_expense_category (              -- 12 หมวด
  expense_cat_id smallint PRIMARY KEY,
  code varchar(10) NOT NULL UNIQUE,
  name_th varchar(100) NOT NULL );               -- เงินเดือน...รายจ่ายอื่น

CREATE TABLE ref_area (
  area_id integer PRIMARY KEY,
  code varchar(10) NOT NULL UNIQUE, name_th varchar(150) NOT NULL,
  gps_lat numeric(10,7), gps_long numeric(10,7) );  -- e-BGIS

-- ===== ชั้น 1–2 =====
CREATE TABLE ministry (
  ministry_id integer PRIMARY KEY,
  code varchar(5) NOT NULL UNIQUE, name_th varchar(255) NOT NULL );   -- 01000

CREATE TABLE agency (
  agency_id integer PRIMARY KEY,
  ministry_id integer NOT NULL REFERENCES ministry,
  code varchar(5) NOT NULL UNIQUE, name_th varchar(255) NOT NULL,     -- 01007
  system_type varchar(5) NOT NULL DEFAULT 'eB' );          -- eB | BBL

-- ===== ชั้น 4: เป้าหมายบริการกระทรวง (Bis65I004) =====
CREATE TABLE ministry_service_target (
  min_target_id bigserial PRIMARY KEY,
  ministry_id integer NOT NULL REFERENCES ministry,
  code varchar(2) NOT NULL,                       -- BC
  /* + คอลัมน์ร่วม */
  UNIQUE (fiscal_year, ministry_id, code) );

-- ===== ชั้น 5: เป้าหมายบริการหน่วยงาน (Bis65I005) =====
CREATE TABLE agency_service_target (
  agy_target_id bigserial PRIMARY KEY,
  agency_id integer NOT NULL REFERENCES agency,
  min_target_id bigint NOT NULL REFERENCES ministry_service_target,  -- M:1
  code varchar(2) NOT NULL,                       -- 07
  UNIQUE (fiscal_year, agency_id, code) );

-- ===== ชั้น 3: แผนงาน =====
CREATE TABLE program (
  program_id bigserial PRIMARY KEY,
  prog_type varchar(20) NOT NULL,    -- บุคลากร|พื้นฐาน|ยุทธศาสตร์|บูรณาการ
  code varchar(20) NOT NULL, UNIQUE (fiscal_year, code) );

-- ===== ชั้น 6: ผลผลิต/โครงการ (Bis65I009) ★ =====
CREATE TABLE output_project (
  output_id bigserial PRIMARY KEY,
  agency_id integer NOT NULL REFERENCES agency,
  agy_target_id bigint NOT NULL REFERENCES agency_service_target,
  program_id bigint NOT NULL REFERENCES program,
  acct_code varchar(7) NOT NULL,     -- XX-XXXX เช่น 00-0244
  op_type varchar(2) NOT NULL,       -- 00=โครงการ|07=บุคลากร|08=กองทุน
  UNIQUE (fiscal_year, agency_id, acct_code) );

-- ===== ชั้น 7: กิจกรรม (Bis65a001) =====
CREATE TABLE activity (
  activity_id bigserial PRIMARY KEY,
  output_id bigint NOT NULL REFERENCES output_project,
  seq_code varchar(10) NOT NULL,     -- 1,2,3
  UNIQUE (fiscal_year, output_id, seq_code) );

-- ===== ชั้น 8: รายการ (Bis65a002) =====
CREATE TABLE item (
  item_id bigserial PRIMARY KEY,
  activity_id bigint NOT NULL REFERENCES activity,
  parent_item_id bigint NULL REFERENCES item,    -- self: ระดับชั้น
  expense_cat_id smallint NOT NULL REFERENCES ref_expense_category,
  area_id integer NULL REFERENCES ref_area,
  item_level smallint NOT NULL DEFAULT 1 CHECK (item_level BETWEEN 1 AND 3),
  code varchar(6) NOT NULL,          -- 000337
  UNIQUE (fiscal_year, code) );

กฎสำคัญ (Business rules)

  1. บังคับลำดับการสร้าง: output ต้องมี agency_service_target ก่อน → activity ต้องมี output → item ต้องมี activity
  2. op_type ต้องตรงกับ prefix ของ acct_code (00-/07-/08-)
  3. ลบ (hard-delete) เฉพาะผู้มีสิทธิ์ + เมื่อไม่มี child (ON DELETE RESTRICT) — ปกติใช้ปิด (CLOSED)
  4. ยกยอดข้ามปี: copy ทั้งสายโซ่จาก fiscal_year เดิม → ปีใหม่
  5. area_id (GPS) บังคับเฉพาะรายการ ครุภัณฑ์/ที่ดิน/สิ่งก่อสร้าง (e-BGIS)
  6. BBL: ใช้ตารางเดียวกัน แต่คุมสิทธิ์แก้ไขที่ application layer ตาม agency.system_type
